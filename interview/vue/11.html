    # 考点十一：vue 的高级特性

    # 1.如何实现自己的 v-model

    //父组件
<CustomModel v-model="a" >

//custom-model组件
<input
    :value="text"
    @input="$emit('change',$event.target.value)"
/>
export default {
    model:{
        props:'text',//对应props text
        event:'change'
    },
    props:{
        text:String,
        default(){
            return '';
        }
    }
}

    # 2.$nextTick

    1. Vue 是异步渲染
    2. data 改变后，DOM 不会立刻渲染
    3. $nextTick 会在 DOM 渲染之后被触发，以获取最新 DOM 节点
    this.$nextTick(() => {
  const ulElem = this.$refs.ul1;
  console.log(ulElem.childNodes.length);
});

    # 3.slot

    1. 基本使用

    2. 作用域插槽

    //子组件
<slot v-bind:user="user">
    {{user.lastName}}
</slot>
//父组件
<template v-slot="slotProps">
    {{slotProps.user.firstName}}
</template>

    # 4.动态组件

    1. :is="component-name"用法
    <component :is="component-name"></component>
    1. 需要根据数据，动态渲染的场景。即组件类型不确定。

    # 5.异步组件

    <FormDemo v-if="布尔值变量" />;

export default {
  components: {
    FormDemo: () => import("../BaseUse/FprmDemo"),
  },
};

    # 6.keep-alive

    1. 缓存组件
    2. 频繁切换，不需要重复渲染
    //第一次渲染会触发生命周期
<keep-alive>
  <Component />
</keep-alive>

    # 7.使用 mixin 抽离 vue 公共逻辑

    1. 多个组件相同的逻辑，抽离出来
    //mixin.js
export default {
  data() {
    return {
      city: "北京",
    };
  },
  methods: {
    showName() {},
  },
  mounted() {},
};

import Mixin from 'mixin';
export default{
    mixins:[Mixin]
}
