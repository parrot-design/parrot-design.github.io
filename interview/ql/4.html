    # 写一个深拷贝

    function deepClone(obj) {
  if (typeof obj !== "object") {
    return obj;
  }
  let result;

  if (Array.isArray(obj)) {
    result = [];
  } else {
    result = {};
  }

  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      result[key] = deepClone(obj[key]);
    }
  }

  return result;
}
